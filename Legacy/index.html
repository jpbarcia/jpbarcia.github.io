            if (c == null)

            Usuario u = db.Usuario.FirstOrDefault(x => x.Id == c.UsuarioId);
            DeudaBinding model = new DeudaBinding()
            {
                id = c.Id,
                deuda = (double)c.Deuda,

                propietarios = u.Propietarios
            };

        }
        [HttpPost]
        public ActionResult Pagar(DeudaBinding model)
        {
            if (!ModelState.IsValid)
            {
                return View(model);
            }
            Cuota c = db.Cuota.FirstOrDefault(x => x.Id == model.id && x.Pagado == false);
            if (c == null)
                return RedirectToAction("DeudasPendientes");
            using (var transaction = db.Database.BeginTransaction())
            {
                try
                {
                    c.Pagado = model.pagado;
                    c.Pago = c.Pago.GetValueOrDefault() + (decimal)model.monto;
                    c.FechaPago = model.fechaPago;
                    if (model.pagado)
                    {
                        Usuario u = db.Usuario.FirstOrDefault(x => x.Id == c.UsuarioId);
                        u.Monto = c.Pago - c.Deuda;
                    }
                    db.SaveChanges();
                    transaction.Commit();
                }
                catch (Exception ex)
                {
                    transaction.Rollback();
                    //Response.StatusCode = 404;
                }
            }
            return View(model);
        }
